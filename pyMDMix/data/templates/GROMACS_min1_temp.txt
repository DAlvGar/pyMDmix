; Parameters describing what to do, when to stop and what to save
integrator	= steep		; Algorithm (steep = stepest descent, better far from minima)
constraints     = none          ; only those described in topology file
emtol		= 100.0  	; Stop minimization when the maximum force < 1000.0 kJ/mol/nm
emstep   	= 0.5	        ; Energy step size

; Parameters describing how to find the neighbors of each atom and how to calculate the interactions
nstlist		= 10            ; Frequency to update the neighbor list and long range forces

rlist		= 1.2		; Cut-off for making neighbor list (short range forces)
coulombtype	= PME		; Treatment of long range electrostatic interactions
rcoulomb	= 1.2		; Short-range electrostatic cut-off
rvdw		= 1.2		; Short-range Van der Waals cut-off
